24 def wait_for_service(service, timeout=None):
25  """
26  Blocks until service is available.
27 
28  Use this in initialization code if your program depends on a service already running.
29 
30  @param service name of service
31  @param timeout timeout time in seconds or Duration, None for no timeout.
32  NOTE: timeout=0 is invalid as wait_for_service actually
33  contacts the service, so non-blocking behavior is not possible.
34  For timeout=0 use cases, just call the service without waiting.
35 
36  @throws ROSException if specified timeout is exceeded
37  @throws ROSInterruptException if shutdown interrupts wait
38  """
39  monostart, timeout = time.monotonic(), ros2.to_sec(timeout)
40  if timeout == 0.:
41  raise ValueError("timeout must be non-zero")
42  deadline = (monostart + timeout) if timeout else 2**31 - 1
43  service, timer, rate, cls, cli = ros2.resolve_name(service), None, None, None, None
44 
45  try:
46  timer = ros2.create_timer(0.1, callback=None)
47  rate = rclpy.timer.Rate(timer, context=ros2.NODE.context)
48 
49  while not cls and time.monotonic() < deadline:
50  typename = next((tt[0] for n, tt in ros2.get_services() if n == service), None)
51  if typename: cls = ros2.get_message_class(typename)
52  else: rate.sleep()
53 
54  mononow = time.monotonic()
55  if mononow >= deadline:
56  raise exceptions.ROSException("timeout exceeded while waiting for service %s" % service)
57 
58  cli = ros2.create_client(service, cls)
59  cli.wait_for_service(None if not timeout else timeout + (mononow - monostart))
60  except rclpy.exceptions.ROSInterruptException as e:
61  raise exceptions.ROSInterruptException(*e.args) from e
62  finally:
63  timer and ros2.destroy_entity(timer)
64  cli and ros2.destroy_entity(cli)
65  timer and ros2.destroy_entity(timer)
