122 def emit(self, record):
123 """Adds message to ROS2 logger."""
124 if not self._logger.is_enabled_for(record.levelno):
125 return
126 try:
127 text = record.msg % record.args if record.args else record.msg
128 except Exception:
129 text = record.msg
130 if record.exc_info:
131 text += "\n\n" + "".join(traceback.format_exception(*record.exc_info))
132 # Cannot use self._logger.log(text, level)
133 # as rclpy raises error if using different levels from same line
134 if logging.DEBUG == record.levelno:
135 self._logger.debug(text)
136 elif logging.INFO == record.levelno:
137 self._logger.info(text)
138 elif logging.WARN == record.levelno:
139 self._logger.warning(text)
140 elif logging.ERROR == record.levelno:
141 self._logger.error(text)
142 elif logging.FATAL == record.levelno:
143 self._logger.fatal(text)
144 else:
145 self._logger.info(text)

264 def get_topic_info(self, counts=False):
265 """
266 Returns topic and message type metainfo as {(topic, typename, typehash): count}.
267
268 @param counts whether to return actual message counts instead of None
269 """
270 DEFAULTCOUNT = 0 if counts else None
271 if not self._counts and self._has_table("topics"):
272 for row in self._db.execute("SELECT * FROM topics ORDER BY name"):
273 topic, typename, typehash = row["name"], canonical(row["type"]), None
274 try:
275 typehash = self._types[typename] if typename in self._types else \
276 get_message_type_hash(typename)
277 except Exception:
278 logger.warning("Error getting message type MD5 hash for %r.",
279 typename, exc_info=True)
280 self._dbtopics[(topic, typename)] = row
281 self._counts[(topic, typename, typehash)] = DEFAULTCOUNT
282 self._types.setdefault(typename, typehash)
283
284 if counts and self._has_table("messages") and not any(self._counts.values()):
285 countkeys = {v["id"]: (t, n, self._types.get(n))
286 for (t, n), v in self._dbtopics.items()}
287 for row in self._db.execute("SELECT topic_id, COUNT(*) AS count FROM messages "
288 "GROUP BY topic_id"):
289 if row["topic_id"] in countkeys:
290 self._counts[countkeys[row["topic_id"]]] = row["count"]
291
292 return dict(self._counts)

310 def median(vals):
311 """Returns median value from given sorted numbers."""
312 vlen = len(vals)
313 return None if not vlen else vals[vlen // 2] if vlen % 2 else \
314 float(vals[vlen // 2 - 1] + vals[vlen // 2]) / 2
315
316 for (t, n, _), c in sorted(counts.items()):
317 if topics and t not in topics: continue # for
318 mymedian = None
319 if c > 1:
320 args = (self._dbtopics[(t, n)]["id"], )
321 cursor = self._db.execute("SELECT timestamp FROM messages WHERE topic_id = ?", args)
322 stamps = sorted(x["timestamp"] / 1E9 for x in cursor)
323 mymedian = median(sorted(s1 - s0 for s1, s0 in zip(stamps[1:], stamps[:-1])))
324 freq = 1.0 / mymedian if mymedian else None
325 topicdict[t] = self.TopicTuple(n, c, len(self.get_qoses(t, n) or []), freq)

359 def read_messages(self, topics=None, start_time=None, end_time=None, raw=False, *_, **__):
360 """
361 Yields messages from the bag in chronological order.
362
363 @param topics list of topics or a single topic to filter by, if at all
364 @param start_time earliest timestamp of message to return,
365 as `rclpy.Time` or convertible
366 (int/float/duration/datetime/decimal/builtin_interfaces.Time)
367 @param end_time latest timestamp of message to return,
368 as `rclpy.Time` or convertible
369 (int/float/duration/datetime/decimal/builtin_interfaces.Time)
370 @param raw if True, then returned messages are tuples of
371 (typename, bytes, typehash, typeclass)
372 @return generator of BagMessage(topic, message, rclpy.time.Time) namedtuples
373 """
374 self.get_topic_info()
375 if not self._dbtopics or (topics is not None and not topics):
376 return
377
378 sql, exprs, args = "SELECT * FROM messages", [], ()
379 if topics:
380 topics = topics if isinstance(topics, (list, set, tuple)) else [topics]
381 topic_ids = [x["id"] for (topic, _), x in self._dbtopics.items() if topic in topics]
382 exprs += ["topic_id IN (%s)" % ", ".join(map(str, topic_ids))]
383 if start_time is not None:
384 exprs += ["timestamp >= ?"]
385 args += (to_nsec(to_time(start_time)), )
386 if end_time is not None:
387 exprs += ["timestamp <= ?"]
388 args += (to_nsec(to_time(end_time)), )
389 sql += ((" WHERE " + " AND ".join(exprs)) if exprs else "")
390 sql += " ORDER BY timestamp"
391
392 topicmap = {v["id"]: v for v in self._dbtopics.values()}
393 msgtypes = {} # {typename: cls}
394 topicset = set(topics or [t for t, _ in self._dbtopics])
395 for row in self._db.execute(sql, args):
396 tdata = topicmap[row["topic_id"]]
397 topic, typename = tdata["name"], canonical(tdata["type"])
398 if not raw and msgtypes.get(typename, typename) is None: continue # for row
399 typehash = self._types.get(typename)
400
401 try:
402 cls = msgtypes[typename] if typename in msgtypes else \
403 msgtypes.setdefault(typename, get_message_class(typename))
404 if raw: msg = (typename, row["data"], typehash, cls)
405 else: msg = rclpy.serialization.deserialize_message(row["data"], cls)
406 except Exception as e:
407 msgtypes.setdefault(typename, None)
408 logger.warning("Error loading type %r in topic %r: %s", typename, topic, e)
409 if raw: msg = (typename, row["data"], typehash, None)
410 elif set(n for n, c in msgtypes.items() if c is None) == topicset:
411 break # for row
412 else: continue # for row
413 stamp = rclpy.time.Time(nanoseconds=row["timestamp"])
414
415 yield self.BagMessage(topic, msg, stamp)
416 if not self._db:
417 break # for row

499 def __str__(self):
500 """Returns informative text for bag, with a full overview of topics and types."""
501
502 indent = lambda s, n: ("\n%s" % (" " * n)).join(s.splitlines())
503 # Returns UNIX timestamp as "Dec 22 2021 23:13:44.44"
504 fmttime = lambda x: datetime.datetime.fromtimestamp(x).strftime("%b %d %Y %H:%M:%S.%f")[:-4]
505 def fmtdur(secs):
506 """Returns duration seconds as text like "1hr 1:12s (3672s)" or "51.8s"."""
507 result = ""
508 hh, rem = divmod(secs, 3600)
509 mm, ss = divmod(rem, 60)
510 if hh: result += "%dhr " % hh
511 if mm: result += "%d:" % mm
512 result += "%ds" % ss
513 if hh or mm: result += " (%ds)" % secs
514 return result
515
516 entries = {}
517 counts = self.get_topic_info(counts=True)
518 start, end = self.get_start_time(), self.get_end_time()
519
520 entries["path"] = self.filename
521 if None not in (start, end):
522 entries["duration"] = fmtdur(end - start)
523 entries["start"] = "%s (%.2f)" % (fmttime(start), start)
524 entries["end"] = "%s (%.2f)" % (fmttime(end), end)
525 entries["size"] = util.format_bytes(self.sizesize)
526 if any(counts.values()):
527 entries["messages"] = str(sum(counts.values()))
528 if counts:
529 nhs = sorted(set((n, h) for _, n, h in counts))
530 namew = max(len(n) for n, _ in nhs)
531 # "pkg/Msg <PAD>[typehash]"
532 entries["types"] = "\n".join("%s%s [%s]" % (n, " " * (namew - len(n)), h) for n, h in nhs)
533 if counts:
534 topicw = max(len(t) for t, _, _ in counts)
535 typew = max(len(n) for _, n, _ in counts)
536 countw = max(len(str(c)) for c in counts.values())
537 lines = []
538 for (t, n, _), c in sorted(counts.items()):
539 qq = self.get_qoses(t, n) or []
540 # "/my/topic<PAD>"
541 line = "%s%s" % (t, " " * (topicw - len(t)))
542 # " <PAD>13 msgs" or " <PAD>1 msg "
543 line += " %s%s msg%s" % (" " * (countw - len(str(c))), c, " " if 1 == c else "s")
544 # " : pkg/Msg"
545 line += " : %s" % n
546 # "<PAD> (2 connections)" if >1 connections
547 line += "%s (%s connections)" % (" " * (typew - len(n)), len(qq)) if len(qq) > 1 else ""
548 lines.append(line)
549 entries["topics"] = "\n".join(lines)
550
551 labelw = max(map(len, entries))
552 return "\n".join("%s:%s %s" % (k, " " * (labelw - len(k)), indent(v, labelw + 2))
553 for k, v in entries.items())

574def init_node(name, args=None, namespace=None, anonymous=False, log_level=None, enable_rosout=True,
575 multithreaded=True, reentrant=False):
576 """
577 Initializes rclpy and creates ROS2 node.
578
579 @param name node name, without namespace
580 @param args list of command-line arguments for the node
581 @param namespace node namespace override
582 @param anonymous whether to auto-generate a unique name for the node,
583 using the given name as base
584 @param log_level level to set for ROS logging
585 (name like "DEBUG" or one of `logging` constants like `logging.DEBUG`)
586 @param enable_rosout `False` to suppress auto-publication of rosout
587 @param multithreaded use `MultiThreadedExecutor` instead of `SingleThreadedExecutor`
588 @param reentrant use `ReentrantCallbackGroup` instead of `MutuallyExclusiveCallbackGroup`
589 @return `rclpy.node.Node`
590 """
591 global NODE, CALLBACK_GROUP, EXECUTOR, SHUTDOWN
592 if NODE: return NODE
593
594 if anonymous:
595 name = "%s_%s_%s" % (name, os.getpid(), int(time.time_ns() * 1000))
596 with Mutex.NODE:
597 if NODE: return NODE
598
599 patch.patch_ros2()
600 logger.debug("Initializing ROS node %r.", name)
601 try: rclpy.init(args=args)
602 except RuntimeError: pass # Raises if called twice at runtime
603 NODE = rclpy.create_node(name, namespace=namespace, enable_rosout=enable_rosout,
604 automatically_declare_parameters_from_overrides=True,
605 allow_undeclared_parameters=True)
606 execls = rclpy.executors.SingleThreadedExecutor
607 grpcls = rclpy.callback_groups.MutuallyExclusiveCallbackGroup
608 if multithreaded: execls = rclpy.executors.MultiThreadedExecutor
609 if reentrant: grpcls = rclpy.callback_groups.ReentrantCallbackGroup
610 EXECUTOR = execls()
611 EXECUTOR.add_node(NODE)
612 defgrp = NODE.default_callback_group
613 CALLBACK_GROUP = defgrp if grpcls is type(defgrp) else grpcls()
614 SHUTDOWN = False
615
616 if not any(isinstance(x, ROSLogHandler) for x in logger.handlers):
617 logger.addHandler(ROSLogHandler(NODE))
618 if log_level is not None:
619 if not isinstance(log_level, str): log_level = logging.getLevelName(log_level)
620 logger.setLevel(log_level)
621 ros_level = rclpy.logging.get_logging_severity_from_string(log_level)
622 NODE.get_logger().set_level(ros_level)
623 return NODE

643 """
644 Declares all parameters on node from defaults dictionary.

775 except KeyboardInterrupt: SPINNER = None # Signal thread to end

784def spin_once(timeout=None):
785 """
786 Executes one pending ROS2 operation or waits until timeout.
787
788 Waits forever if timeout not given or negative.
789
790 @param timeout time to wait at most, as seconds or ROS2 duration;
791 None or <0 waits forever
792 """
793 global SPINNER
794 timeout = to_sec(timeout)
795 if SPINNER or Mutex.SPIN_ONCE._is_owned(): # Executor already spinning: sleep instead
796 timeout = 2**31 - 1 if timeout is None or timeout < 0 else timeout
797 group = rclpy.callback_groups.ReentrantCallbackGroup()
798 timer = NODE.create_timer(timeout, callback=None, callback_group=group)
799 rclpy.timer.Rate(timer, context=NODE.context).sleep()
800 NODE.destroy_timer(timer)
801 return
802 with Mutex.SPIN_ONCE:
803 EXECUTOR.spin_once(timeout_sec=timeout)

806def spin_until_future_complete(future, timeout=None):
807 """
808 Spins ROS2 until future complete or timeout reached or ROS shut down.
809
810 @param future object with `concurrent.futures.Future`-conforming interface to complete
811 @param timeout time to wait, as seconds or ROS2 duration
812 """
813 if future.done():
814 return
815 timeout = to_sec(timeout)
816 if timeout == 0.0:
817 if not future.done():
818 spin_once(1E-9)
819 return
820
821 now = time.monotonic()
822 deadline = (now + timeout) if timeout is not None and timeout >= 0 else -1
823 while ok() and (deadline < 0 or now < deadline) and not future.done():
824 spin_once(0.1)
825 now = time.monotonic()
826 if not future.done() and not ok():
827 future.cancel("ROS shut down")

1014def get_topic_qos(topic, cls_or_typename, queue_size=10, publish=False):
1015 """
1016 Returns dictionary for populating rclpy.qos.QoSProfile compatible with counterparts on topic.
1017
1018 @param topic topic full name

1020 @param queue_size populates "depth"

1033 if rels: args.update(reliability=max(rels)) # DEFAULT < RELIABLE < BEST_EFFORT

1119@util.memoize
1120def canonical(typename):
1121 """
1122 Returns "pkg/Type" for "pkg/msg/Type", standardizes various ROS2 formats.
1123
1124 Converts DDS types like "octet" to "byte", and "sequence<uint8, 100>" to "uint8[100]".
1125 """
1126 is_array, bound, dimension = False, "", ""
1127
1128 if "<" in typename:
1129 match = re.match("sequence<(.+)>", typename)
1130 if match: # "sequence<uint8, 100>" or "sequence<uint8>"
1131 is_array = True
1132 typename = match.group(1)
1133 match = re.match(r"([^,]+)?,\s?(\d+)", typename)
1134 if match: # sequence<uint8, 10>
1135 typename = match.group(1)
1136 if match.lastindex > 1: dimension = match.group(2)
1137
1138 match = re.match("(w?string)<(.+)>", typename)
1139 if match: # string<5>
1140 typename, bound = match.groups()
1141
1142 if "[" in typename: # "string<=5[<=10]" or "string<=5[10]" or "byte[10]" or "byte[]"
1143 dimension = typename[typename.index("[") + 1:typename.index("]")]
1144 typename, is_array = typename[:typename.index("[")], True
1145
1146 if "<=" in typename: # "string<=5"
1147 typename, bound = typename.split("<=")
1148
1149 if typename.count("/") > 1:
1150 typename = "%s/%s" % tuple((x[0], x[-1]) for x in [typename.split("/")])[0]
1151
1152 suffix = ("<=%s" % bound if bound else "") + ("[%s]" % dimension if is_array else "")
1153 return DDS_TYPES.get(typename, typename) + suffix
