32def init_node(name, args=None, namespace=None, anonymous=False, log_level=None, enable_rosout=True,
33 multithreaded=True, reentrant=False):
34 """
35 Initializes ROS and creates ROS node.
36
37 @param name node name, without namespace
38 @param args list of command-line arguments for the node
39 @param namespace node namespace override
40 @param anonymous whether to auto-generate a unique name for the node,
41 using the given name as base
42 @param log_level level to set for ROS logging
43 (name like "DEBUG" or one of `logging` constants like `logging.DEBUG`)
44 @param enable_rosout `False` to suppress auto-publication of rosout
45 @param multithreaded ROS2 only, ignored in ROS1:
46 use `MultiThreadedExecutor` instead of `SingleThreadedExecutor`
47 @param reentrant ROS2 only, ignored in ROS1:
48 use `ReentrantCallbackGroup` instead of `MutuallyExclusiveCallbackGroup`
49 @return `None` in ROS1, `rclpy.node.Node` in ROS2
50 """
51 if ros1: args = (name, args, namespace, anonymous, log_level, enable_rosout)
52 else: args = (name, args, namespace, anonymous, log_level, multithreaded, reentrant)
53 return (ros1.init_node if ros1 else ros2.init_node)(*args)

174def create_publisher(topic, cls_or_typename, latch=False, queue_size=0, **qosargs):
175 """
176 Returns a ROS publisher instance.
177
178 @param topic name of topic to open
179 @param cls_or_typename ROS message class object like `std_msgs.msg.Bool`
180 or message type name like "std_msgs/Bool"
181 @param latch provide last published message to new subscribers
182 (sets `DurabilityPolicy.TRANSIENT_LOCAL` in ROS2)
183 @param queue_size queue size of outgoing messages (0 or None: infinite)
184 (sets QoS `depth` in ROS2)
185 @param qosargs additional key-value arguments for ROS2
186 `QoSProfile`, like `reliability` (uses only `depth`
187 and `durability`=TRANSIENT_LOCAL in ROS1)
188 @return `rospy.Publisher` or `rclpy.publisher.Publisher`,
189 either will support keyword arguments in `publish()`
190 and have `get_num_connections()`
191 """
192 if ros1:
193 queue_size = qosargs.get("depth", queue_size)
194 if qosargs.get("durability") == 1: # DurabilityPolicy.TRANSIENT_LOCAL
195 latch = True
196 return ros1.create_publisher(topic, cls_or_typename, latch, queue_size)
197 return ros2.create_publisher(topic, cls_or_typename, latch, queue_size, **qosargs)

200def create_subscriber(topic, cls_or_typename, callback, callback_args=None,
201 queue_size=0, raw=False, **qosargs):
202 """
203 Returns a ROS subscriber instance.
204
205 @param topic name of topic to listen to
206 @param cls_or_typename ROS message class object like `std_msgs.msg.Bool`
207 or message type name like "std_msgs/Bool"
208 @param callback callback function, invoked with received message,
209 and with additional arguments if given
210 @param callback_args additional arguments to pass to the callback, if any,
211 invoked as `callback(msg, callback_args)Â´
212 @param queue_size queue size of incoming messages (0 or None: infinite)
213 @param raw invoke callback with serialized message bytes,
214 make subscription with AnyMsg in ROS1
215 @param qosargs additional key-value arguments for ROS2
216 `QoSProfile`, like `reliability` (uses only `depth` in ROS1).
217 `__autodetect` will look up current publishers on the topic
218 and create a compatible QoS.
219 @return `rospy.Subscriber` or `rclpy.subscription.Subscription`
220 """
221 if ros1:
222 queue_size = qosargs.get("depth", queue_size)
223 return ros1.create_subscriber(topic, cls_or_typename, callback, callback_args,
224 queue_size, raw)
225 return ros2.create_subscriber(topic, cls_or_typename, callback, callback_args,
226 queue_size, raw, **qosargs)
