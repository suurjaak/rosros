261 def dict_to_message(dct, msg):
262  """
263  Returns given ROS message populated from Python dictionary.
264 
265  Raises TypeError on attribute value type mismatch.
266  """
267  for name, typename in ros.get_message_fields(msg).items():
268  if name not in dct:
269  continue # for name,
270  v, msgv = dct[name], ros.get_message_value(msg, name)
271 
272  if ros.is_ros_message(msgv):
273  v = dict_to_message(v, msgv)
274  elif isinstance(msgv, (list, tuple)):
275  scalarname = ros.scalar(typename)
276  if scalarname in ROS_BUILTIN_TYPES:
277  cls = ROS_BUILTIN_CTORS[scalarname]
278  v = [x if isinstance(x, cls) else cls(x) for x in v]
279  else:
280  cls = ros.get_message_class(scalarname)
281  v = [dict_to_message(x, cls()) for x in v]
282  else:
283  v = type(msgv)(v)
284 
285  setattr(msg, name, v)
286  return msg

289 def message_to_dict(msg, replace=None):
290  """
291  Returns ROS message as nested Python dictionary.
292 
293  @param replace mapping of {value: replaced value},
294  e.g. {math.nan: None, math.inf: None}
295  """
296  result = {} if ros.is_ros_message(msg) else msg
297  for name, typename in ros.get_message_fields(msg).items():
298  v = ros.get_message_value(msg, name)
299  if ros.is_ros_time(v):
300  v = dict(zip(get_message_fields(v), ros.to_sec_nsec(v)))
301  elif ros.is_ros_message(v):
302  v = message_to_dict(v)
303  elif isinstance(v, (list, tuple)):
304  if ros.scalar(typename) not in ROS_BUILTIN_TYPES:
305  v = [message_to_dict(x) for x in v]
306  elif replace:
307  v = [replace.get(x, x) for x in v]
308  elif replace:
309  v = replace.get(v, v)
310  result[name] = v
311  return result

314 def message_to_str(msg, indent=None):
315  """
316  Returns ROS message as an evaluatable string, e.g. "std_msgs.msg.UInt8(data=0)".
317 
318  @param indent multi-line indentation level to use if not returning one-liner;
319  as the number of spaces or the string to indent with
320  """
321  if not is_ros_message(msg):
322  return repr(msg)
323  indent = " " * indent if isinstance(indent, int) else indent or ""
324  sep, start, end = (",\n", "\n%s" % indent, "\n") if indent else (", ", "", "")
325  parts = []
326  for k in get_message_fields(msg):
327  v = get_message_value(msg, k)
328  if is_ros_message(v):
329  v = message_to_str(v, indent)
330  if indent:
331  v = v.replace("\n", start)
332  elif isinstance(v, bytes):
333  v = "bytes(%s)" % list(v)
334  elif isinstance(v, list) and v:
335  nested = sep.join(message_to_str(x, indent) for x in v)
336  if indent:
337  nested = indent + nested.replace("\n", "\n%s" % (indent * 2))
338  v = "[%s%s%s%s]" % (start, nested, end, indent)
339  else:
340  v = repr(v)
341  parts.append("%s=%s" % (k, v))
342  name = make_full_typename(get_message_type(msg)).replace("/", ".")
343  return "%s(%s%s%s)" % (name, start, (sep + indent).join(parts), end)
