456  def rostimer_sleep_on_event(duration, event=None):
457  """
458  Sleeps for the specified duration in ROS time, or until event gets set.
459 
460  Returns immediately if duration is negative or event is set.
461 
462  @param duration seconds (or rospy.Duration) to sleep
463  @param event threading.Event to wait() on, if any,
464  will return early if event gets set
465 
466  @throws ROSInterruptException if ROS shutdown occurs before sleep completes
467  @throws ROSTimeMovedBackwardsException if ROS time is set backwards
468  """
469  if event and event.is_set(): return
470 
471  if rospy.rostime.is_wallclock():
472  if isinstance(duration, genpy.Duration):
473  duration = duration.to_sec()
474  if duration < 0:
475  return
476  event.wait(duration) if event else rospy.rostime.wallsleep(duration)
477  else:
478  initial_rostime = rospy.rostime.get_rostime()
479  if not isinstance(duration, genpy.Duration):
480  duration = genpy.Duration.from_sec(duration)
481 
482  rostime_cond = rospy.rostime.get_rostime_cond()
483 
484  if initial_rostime == genpy.Time(0):
485  # break loop if time is initialized or node is shutdown
486  while initial_rostime == genpy.Time(0) and \
487  not rospy.core.is_shutdown():
488  with rostime_cond:
489  rostime_cond.wait(0.3)
490  initial_rostime = rospy.rostime.get_rostime()
491  if event and event.is_set(): return
492 
493  sleep_t = initial_rostime + duration
494 
495  # break loop if sleep_t is reached, time moves backwards, or node is shutdown
496  while rospy.rostime.get_rostime() < sleep_t \
497  and rospy.rostime.get_rostime() >= initial_rostime and not rospy.core.is_shutdown():
498  with rostime_cond:
499  rostime_cond.wait(0.5)
500  if event and event.is_set(): return
501 
502  if rospy.rostime.get_rostime() < initial_rostime:
503  time_jump = (initial_rostime - rospy.rostime.get_rostime()).to_sec()
504  raise rospy.exceptions.ROSTimeMovedBackwardsException(time_jump)
505  if rospy.core.is_shutdown():
506  raise rospy.exceptions.ROSInterruptException("ROS shutdown request")
