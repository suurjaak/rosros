478  def rostimer_sleep_on_event(duration, event=None):
479  """
480  Sleeps for the specified duration in ROS time, or until event gets set.
481 
482  Returns immediately if duration is negative or event is set.
483 
484  @param duration seconds (or rospy.Duration) to sleep
485  @param event threading.Event to wait() on, if any,
486  will return early if event gets set
487 
488  @throws ROSInterruptException if ROS shutdown occurs before sleep completes
489  @throws ROSTimeMovedBackwardsException if ROS time is set backwards
490  """
491  if event and event.is_set(): return
492 
493  if rospy.rostime.is_wallclock():
494  if isinstance(duration, genpy.Duration):
495  duration = duration.to_sec()
496  if duration < 0:
497  return
498  event.wait(duration) if event else rospy.rostime.wallsleep(duration)
499  else:
500  initial_rostime = rospy.rostime.get_rostime()
501  if not isinstance(duration, genpy.Duration):
502  duration = genpy.Duration.from_sec(duration)
503 
504  rostime_cond = rospy.rostime.get_rostime_cond()
505 
506  if initial_rostime == genpy.Time(0):
507  # break loop if time is initialized or node is shutdown
508  while initial_rostime == genpy.Time(0) and \
509  not rospy.core.is_shutdown():
510  with rostime_cond:
511  rostime_cond.wait(0.3)
512  initial_rostime = rospy.rostime.get_rostime()
513  if event and event.is_set(): return
514 
515  sleep_t = initial_rostime + duration
516 
517  # break loop if sleep_t is reached, time moves backwards, or node is shutdown
518  while rospy.rostime.get_rostime() < sleep_t \
519  and rospy.rostime.get_rostime() >= initial_rostime and not rospy.core.is_shutdown():
520  with rostime_cond:
521  rostime_cond.wait(0.5)
522  if event and event.is_set(): return
523 
524  if rospy.rostime.get_rostime() < initial_rostime:
525  time_jump = (initial_rostime - rospy.rostime.get_rostime()).to_sec()
526  raise rospy.exceptions.ROSTimeMovedBackwardsException(time_jump)
527  if rospy.core.is_shutdown():
528  raise rospy.exceptions.ROSInterruptException("ROS shutdown request")
