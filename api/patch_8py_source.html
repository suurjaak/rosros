367 def Timer__init(self, period, callback, oneshot=False, reset=False):
368 """
369 Constructs a new ROS1 Timer.
370
371 Plugs in ROS2 API compatibility: cancelling, resetting, and public timing info.
372
373 @param period desired period between callbacks, as rospy.Duration
374 @param callback callback to be called, taking rospy.TimerEvent
375 @param oneshot if True, fire only once, otherwise fire continuously
376 until shutdown is called [default: False]
377 @param reset if True, timer is reset when rostime moved backward. [default: False]
378 """
379 self._paused = False
380 self._wakeable = threading.Event() # Interruptable sleeper
381 self._time_last = None # rospy.Time() of start or last call or reset
382 self._time_next = None # rospy.Time() of next expected call
383 self._clock = Clock()
384 ROS1_Timer__init(self, period, callback, oneshot, reset) # Starts thread
385
386 def Timer__shutdown(self):
387 """Stop firing callbacks."""
388 self._shutdown = True
389 self._wakeable.set()
390
391 def Timer__run(self):
392 """Override rospy.Timer.run() to support canceling and resetting, and call timestamps."""
393 self._time_last = self._clock.now()
394 self._time_next = self._time_last + self._period
395 last_expected, last_real, last_duration = None, None, None
396 self._wakeable.clear()
397 while not rospy.core.is_shutdown() and not self._shutdown:
398 try:
399 rostimer_sleep_on_event(self._period.to_sec(), event=self._wakeable)
400 except rospy.exceptions.ROSTimeMovedBackwardsException:
401 if not self._reset:
402 raise
403 except rospy.exceptions.ROSInterruptException:
404 if rospy.core.is_shutdown():
405 break
406 raise
407 if self._shutdown:
408 break
409 if self._paused:
410 self._wakeable.wait()
411 if self._shutdown:
412 break
413 if self._paused is None:
414 self._paused = False
415 self._time_last = self._clock.now()
416 self._time_next = self._time_last + self._period
417 last_expected, last_real, last_duration = None, None, None
418 self._wakeable.clear()
419 continue
420 current_real = self._clock.now()
421 start = time.time()
422 self._time_last = self._clock.now()
423 self._callback(rospy.timer.TimerEvent(last_expected, last_real, self._time_next,
424 current_real, last_duration))
425 if self._oneshot:
426 break
427 last_duration = time.time() - start
428 last_expected, last_real = self._time_next, current_real
429 self._time_next += self._period

478 def rostimer_sleep_on_event(duration, event=None):
479 """
480 Sleeps for the specified duration in ROS time, or until event gets set.
481
482 Returns immediately if duration is negative or event is set.
483
484 @param duration seconds (or rospy.Duration) to sleep
485 @param event threading.Event to wait() on, if any,
486 will return early if event gets set
487
488 @throws ROSInterruptException if ROS shutdown occurs before sleep completes
489 @throws ROSTimeMovedBackwardsException if ROS time is set backwards
490 """
491 if event and event.is_set(): return
492
493 if rospy.rostime.is_wallclock():
494 if isinstance(duration, genpy.Duration):
495 duration = duration.to_sec()
496 if duration < 0:
497 return
498 event.wait(duration) if event else rospy.rostime.wallsleep(duration)
499 else:
500 initial_rostime = rospy.rostime.get_rostime()
501 if not isinstance(duration, genpy.Duration):
502 duration = genpy.Duration.from_sec(duration)
503
504 rostime_cond = rospy.rostime.get_rostime_cond()
505
506 if initial_rostime == genpy.Time(0):
507 # break loop if time is initialized or node is shutdown
508 while initial_rostime == genpy.Time(0) and \
509 not rospy.core.is_shutdown():
510 with rostime_cond:
511 rostime_cond.wait(0.3)
512 initial_rostime = rospy.rostime.get_rostime()
513 if event and event.is_set(): return
514
515 sleep_t = initial_rostime + duration
516
517 # break loop if sleep_t is reached, time moves backwards, or node is shutdown
518 while rospy.rostime.get_rostime() < sleep_t \
519 and rospy.rostime.get_rostime() >= initial_rostime and not rospy.core.is_shutdown():
520 with rostime_cond:
521 rostime_cond.wait(0.5)
522 if event and event.is_set(): return
523
524 if rospy.rostime.get_rostime() < initial_rostime:
525 time_jump = (initial_rostime - rospy.rostime.get_rostime()).to_sec()
526 raise rospy.exceptions.ROSTimeMovedBackwardsException(time_jump)
527 if rospy.core.is_shutdown():
528 raise rospy.exceptions.ROSInterruptException("ROS shutdown request")

729 @param cb_cargs additional arguments associated with callback
