25def sleep(duration):
26 """
27 Sleeps for the specified duration in ROS time.
28
29 @param duration seconds (or Duration) to sleep, <=0 returns immediately
30
31 @throws ROSInterruptException if ROS shutdown occurs before sleep completes
32 @throws ROSTimeMovedBackwardsException if ROS time is set backwards
33 """
34 jumpback = None
35 timer = ros2.create_timer(ros2.to_sec(duration), callback=None)
36
37 def jumphandler(jump):
38 nonlocal jumpback
39 jumpback = jump
40 ros2.destroy_entity(rate)
41
42 thresh = rclpy.clock.JumpThreshold(min_forward=None, min_backward=ros2.make_duration(nsecs=-1))
43 handle = ros2.NODE.get_clock().create_jump_callback(thresh, post_callback=jumphandler)
44 rate = rclpy.timer.Rate(timer, context=ros2.NODE.context)
45 try:
46 rate.sleep()
47 except rclpy.exceptions.ROSInterruptException as e:
48 raise exceptions.ROSInterruptException(*e.args) from e
49 finally:
50 handle.unregister()
51 ros2.destroy_entity(timer)
52 if jumpback:
53 raise exceptions.ROSTimeMovedBackwardsException(ros2.to_sec(jumpback.delta))

98 def __new__(cls, period, callback, oneshot=False, reset=None):
99 """
100 Returns rclpy.timer.Timer instance.
101
102 @param period desired period between callbacks, as `Duration`
103 @param callback function to be called, taking rospy.TimerEvent
104 @param oneshot if True, fire only once, otherwise fire continuously
105 until shutdown is called [default: False]
106 @param reset ignored (ROS1 compatibility stand-in)
107 """
108 if callable(callback):
109 def callwrapper(period, callback):
110 """Closure for timer callback, retains state for populating TimerEvent."""
111 if isinstance(period, (int, float)): period = ros2.make_duration(period)
112 last_expected, last_real, last_duration = None, None, None
113 current_expected = ros2.get_rostime() + period
114
115 def inner():
116 """Feeds TimerEvent to callback, expected in rospy API."""
117 nonlocal current_expected, last_expected, last_real, last_duration
118 current_real = ros2.get_rostime()
119 start = time.time()
120 callback(TimerEvent(last_expected, last_real, current_expected, current_real,
121 last_duration))
122 last_duration = time.time() - start
123 last_expected, last_real = current_expected, current_real
124 current_expected += period
125 return inner
126 callback = callwrapper(period, callback)
